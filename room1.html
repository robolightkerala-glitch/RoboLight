Production-Grade Firmware Code

Advanced_Ultrasonic_Radar.ino 
/*
 * ROBOLIGHT Advanced Ultrasonic Distance Radar
 * Architecture: Non-blocking state processing with Moving Average Filter
 */

// Pin Assignments
const int TRIG_PIN  = 9;
const int ECHO_PIN  = 10;
const int LED_GREEN = 3;  // PWM Capable
const int LED_YELLOW= 5;  // PWM Capable
const int LED_RED   = 6;  // PWM Capable

// Filtering Parameters
const int FILTER_SAMPLES = 5;
int readBuffer[FILTER_SAMPLES];
int bufferIndex = 0;

// Non-blocking Timing Variables
unsigned long previousMillis = 0;
const long SAMPLE_INTERVAL   = 60; // Sensor pulse cycle interval (ms)

void setup() {
  Serial.begin(115200);
  
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  
  pinMode(LED_GREEN, OUTPUT);
  pinMode(LED_YELLOW, OUTPUT);
  pinMode(LED_RED, OUTPUT);

  // Initialize filter buffer
  for (int i = 0; i < FILTER_SAMPLES; i++) {
    readBuffer[i] = 0;
  }
}

void loop() {
  unsigned long currentMillis = millis();

  // Execute non-blocking sensor acquisition every 60 ms
  if (currentMillis - previousMillis >= SAMPLE_INTERVAL) {
    previousMillis = currentMillis;

    int rawDistance = measureDistanceCm();
    int filteredDistance = applyMovingAverage(rawDistance);

    updateDistanceZones(filteredDistance);
    
    // Telemetry output
    Serial.print("Raw: ");
    Serial.print(rawDistance);
    Serial.print(" cm | Filtered: ");
    Serial.print(filteredDistance);
    Serial.println(" cm");
  }
}

// Low-level hardware pulse calculation
int measureDistanceCm() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  long duration = pulseIn(ECHO_PIN, HIGH, 30000); // 30ms timeout
  if (duration == 0) return 400; // Return out-of-range value on timeout
  
  return duration * 0.0343 / 2; // Speed of sound calculation
}

// Digital moving average filter to remove false spikes
int applyMovingAverage(int newValue) {
  readBuffer[bufferIndex] = newValue;
  bufferIndex = (bufferIndex + 1) % FILTER_SAMPLES;

  long sum = 0;
  for (int i = 0; i < FILTER_SAMPLES; i++) {
    sum += readBuffer[i];
  }
  return sum / FILTER_SAMPLES;
}

// Zone decision engine & adaptive PWM LED driver
void updateDistanceZones(int distanceCm) {
  if (distanceCm > 30) {
    // Green Zone
    analogWrite(LED_GREEN, 255);
    analogWrite(LED_YELLOW, 0);
    analogWrite(LED_RED, 0);
  } 
  else if (distanceCm >= 15 && distanceCm <= 30) {
    // Yellow Warning Zone - Dynamic Dimming
    int pwmValue = map(distanceCm, 15, 30, 255, 50);
    analogWrite(LED_GREEN, 0);
    analogWrite(LED_YELLOW, pwmValue);
    analogWrite(LED_RED, 0);
  } 
  else {
    // Red Danger Zone
    analogWrite(LED_GREEN, 0);
    analogWrite(LED_YELLOW, 0);
    analogWrite(LED_RED, 255);
  }
}

Execution Procedure

  1. Breadboard Power Setup

    Connect the 5V and GND pins from your Arduino directly to the power rails of the breadboard to supply electricity to all modules.

  2. Sensor Mount & Signal Wiring

    Insert the HC-SR04 module into the breadboard. Run jumpers from VCC to 5V rail, GND to GND rail, Trig to Digital Pin 9, and Echo to Digital Pin 10.

  3. LED Output Array Wiring

    Place Green, Yellow, and Red LEDs on the board. Attach a 220Ω resistor from each LED cathode (short leg) to Ground. Connect Green Anode to Pin 3, Yellow Anode to Pin 5, and Red Anode to Pin 6.

  4. Firmware Deployment

    Copy the code snippet above into your Arduino IDE, set the baud rate to 115200, compile, and upload to your board.
